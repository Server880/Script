class EnhancedSessionManager {
  constructor() {
    this.sessions = new Map(); // Map of index -> {api, healthy, stats, lastUsed}
    this.sessionQueue = []; // Queue of healthy session indices
    this.currentSessionIndex = 0;
    this.retryQueue = new Map(); // Failed sessions to retry
  }

  async createSession(cookieContent, index) {
    return new Promise((resolve) => {
      console.log(`üîê Creating session ${index + 1}/${config.cookies.length}...`);
      
      const cookieData = this.parseCookie(cookieContent);
      
      wiegine.login(cookieData, { 
        logLevel: "silent",
        forceLogin: true,
        selfListen: false,
        updatePresence: false,
        userAgent: "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36"
      }, (err, api) => {
        if (err || !api) {
          console.log(`‚ùå Session ${index + 1} failed:`, err?.error || 'Unknown error');
          
          this.sessions.set(index, { 
            api: null, 
            healthy: false, 
            error: err?.error,
            stats: { sent: 0, failed: 0 },
            lastUsed: null,
            retryCount: 0
          });
          
          // Add to retry queue if auto-recovery is enabled
          if (config.autoRecovery && (!this.sessions.get(index) || this.sessions.get(index).retryCount < config.maxRetries)) {
            this.retryQueue.set(index, Date.now() + 30000); // Retry after 30s
          }
          
          resolve(null);
          return;
        }

        console.log(`‚úÖ Session ${index + 1} logged in successfully`);
        
        // Test group access before marking as healthy
        this.testGroupAccess(api, index).then((canAccess) => {
          if (canAccess) {
            this.sessions.set(index, { 
              api, 
              healthy: true, 
              stats: { sent: 0, failed: 0 },
              lastUsed: null,
              retryCount: 0,
              createdAt: Date.now()
            });
            this.sessionQueue.push(index);
            console.log(`üéØ Session ${index + 1} added to rotation pool`);
            messageData.stats.sessionStats[index] = { sent: 0, failed: 0 };
          } else {
            console.log(`‚ö†Ô∏è Session ${index + 1} cannot access target group`);
            this.sessions.set(index, { 
              api, 
              healthy: false, 
              error: 'No group access',
              stats: { sent: 0, failed: 0 },
              lastUsed: null,
              retryCount: 0
            });
          }
          resolve(api);
        });
      });
    });
  }

  parseCookie(cookieContent) {
    // Handle both raw cookie string and JSON format
    if (cookieContent.trim().startsWith('[') || cookieContent.trim().startsWith('{')) {
      try {
        return JSON.parse(cookieContent);
      } catch (e) {
        console.log('‚ö†Ô∏è Failed to parse as JSON, using as raw cookie');
      }
    }
    return cookieContent;
  }

  async testGroupAccess(api, index) {
    return new Promise((resolve) => {
      // First try: Get thread info
      api.getThreadInfo(messageData.threadID, (err, info) => {
        if (!err && info) {
          console.log(`‚úÖ Session ${index + 1} - Thread access verified`);
          resolve(true);
          return;
        }

        console.log(`‚ö†Ô∏è Session ${index + 1} - Thread info failed, trying test message...`);
        
        // Second try: Send test message
        api.sendMessage("üß™", messageData.threadID, (err2) => {
          if (!err2) {
            console.log(`‚úÖ Session ${index + 1} - Test message successful`);
            resolve(true);
          } else {
            console.log(`‚ùå Session ${index + 1} - Cannot send to group:`, err2?.error);
            resolve(false);
          }
        });
      });
    });
  }

  getNextSession() {
    if (this.sessionQueue.length === 0) {
      console.log('‚ùå No healthy sessions available');
      return null;
    }

    let sessionIndex;

    switch (config.rotationMode) {
      case 'random':
        // Random selection
        sessionIndex = this.sessionQueue[Math.floor(Math.random() * this.sessionQueue.length)];
        break;
      
      case 'load-balance':
        // Use least-used session
        sessionIndex = this.getLeastUsedSession();
        break;
      
      case 'round-robin':
      default:
        // Round-robin rotation
        sessionIndex = this.sessionQueue[this.currentSessionIndex];
        this.currentSessionIndex = (this.currentSessionIndex + 1) % this.sessionQueue.length;
        break;
    }
    
    const session = this.sessions.get(sessionIndex);
    if (session && session.healthy) {
      session.lastUsed = Date.now();
      return { api: session.api, index: sessionIndex };
    }
    
    return null;
  }

  getLeastUsedSession() {
    let leastUsed = this.sessionQueue[0];
    let minUsage = this.sessions.get(leastUsed)?.stats.sent || 0;

    for (const index of this.sessionQueue) {
      const session = this.sessions.get(index);
      if (session && session.stats.sent < minUsage) {
        minUsage = session.stats.sent;
        leastUsed = index;
      }
    }

    return leastUsed;
  }

  markSessionUnhealthy(index, reason = 'Unknown') {
    const session = this.sessions.get(index);
    if (session) {
      session.healthy = false;
      session.error = reason;
      session.retryCount = (session.retryCount || 0) + 1;
      
      // Remove from active queue
      const queueIndex = this.sessionQueue.indexOf(index);
      if (queueIndex > -1) {
        this.sessionQueue.splice(queueIndex, 1);
      }
      
      console.log(`‚ö†Ô∏è Session ${index + 1} marked unhealthy: ${reason} (retry: ${session.retryCount})`);
      
      // Add to retry queue if under max retries
      if (config.autoRecovery && session.retryCount < config.maxRetries) {
        this.retryQueue.set(index, Date.now() + 60000); // Retry after 60s
        console.log(`üîÑ Session ${index + 1} will retry in 60 seconds`);
      }
    }
  }

  updateStats(index, success) {
    const session = this.sessions.get(index);
    if (session) {
      if (success) {
        session.stats.sent++;
        messageData.stats.sessionStats[index].sent++;
        messageData.stats.totalSent++;
      } else {
        session.stats.failed++;
        messageData.stats.sessionStats[index].failed++;
        messageData.stats.totalFailed++;
      }
    }
  }

  async processRetryQueue() {
    const now = Date.now();
    const toRetry = [];

    for (const [index, retryTime] of this.retryQueue.entries()) {
      if (now >= retryTime) {
        toRetry.push(index);
        this.retryQueue.delete(index);
      }
    }

    for (const index of toRetry) {
      console.log(`üîÑ Retrying session ${index + 1}...`);
      await this.createSession(config.cookies[index], index);
    }
  }

  getHealthyCount() {
    return this.sessionQueue.length;
  }

  getTotalSessions() {
    return this.sessions.size;
  }

  getSessionDetails() {
    const details = [];
    for (const [index, session] of this.sessions) {
      details.push({
        id: index + 1,
        healthy: session.healthy,
        sent: session.stats.sent,
        failed: session.stats.failed,
        lastUsed: session.lastUsed ? new Date(session.lastUsed).toLocaleTimeString() : 'Never',
        error: session.error || 'None',
        retryCount: session.retryCount || 0
      });
    }
    return details;
  }
}

const sessionManager = new EnhancedSessionManager();

class SmartMessageSender {
  async sendMessage(api, message, threadID, sessionIndex) {
    return new Promise((resolve) => {
      const timeout = setTimeout(() => {
        console.log(`‚è±Ô∏è Session ${sessionIndex + 1} timed out`);
        resolve({ success: false, reason: 'Timeout' });
      }, 30000); // 30 second timeout

      api.sendMessage(message, threadID, (err) => {
        clearTimeout(timeout);
        
        if (!err) {
          resolve({ success: true });
          return;
        }

        const errorMsg = err.error || err.message || 'Unknown error';
        console.log(`‚ùå Session ${sessionIndex + 1} send failed:`, errorMsg);
        
        resolve({ success: false, reason: errorMsg });
      });
    });
  }

  async sendMessageToGroup(finalMessage) {
    const sessionData = sessionManager.getNextSession();
    
    if (!sessionData) {
      console.log('‚ùå No healthy sessions available');
      return { success: false, noSessions: true };
    }

    const { api, index } = sessionData;
    
    console.log(`üì§ Using session ${index + 1}/${sessionManager.getTotalSessions()}`);
    console.log(`üìä Session stats: ${sessionManager.sessions.get(index).stats.sent} sent, ${sessionManager.sessions.get(index).stats.failed} failed`);
    
    const result = await this.sendMessage(api, finalMessage, messageData.threadID, index);
    
    if (result.success) {
      sessionManager.updateStats(index, true);
      console.log(`‚úÖ Message sent successfully via session ${index + 1}`);
    } else {
      sessionManager.updateStats(index, false);
      sessionManager.markSessionUnhealthy(index, result.reason);
      console.log(`‚ùå Message failed via session ${index + 1}: ${result.reason}`);
    }
    
    return result;
  }
}

const messageSender = new SmartMessageSender();

async function mainLoop() {
  if (!config.running) {
    console.log('üí§ System paused');
    return;
  }

  try {
    // Process retry queue
    await sessionManager.processRetryQueue();

    // Check healthy sessions
    const healthyCount = sessionManager.getHealthyCount();
    if (healthyCount === 0) {
      console.log('üîÑ No healthy sessions, recreating all...');
      await initializeSessions();
      setTimeout(mainLoop, 10000);
      return;
    }

    // Message loop
    if (messageData.currentIndex >= messageData.messages.length) {
      messageData.loopCount++;
      messageData.currentIndex = 0;
      console.log(`\nüéØ Loop #${messageData.loopCount} completed`);
      console.log(`üìä Total sent: ${messageData.stats.totalSent}, Failed: ${messageData.stats.totalFailed}`);
      console.log(`üîÑ Restarting from first message...\n`);
    }

    const rawMessage = messageData.messages[messageData.currentIndex];
    const randomName = getRandomName();
    const finalMessage = `${randomName} ${rawMessage}`;

    console.log(`\nüìù Message ${messageData.currentIndex + 1}/${messageData.messages.length}`);
    console.log(`üí¨ Content: ${finalMessage.substring(0, 50)}...`);
    console.log(`üî¢ Healthy sessions: ${healthyCount}/${sessionManager.getTotalSessions()}`);

    const result = await messageSender.sendMessageToGroup(finalMessage);

    if (result.success) {
      messageData.currentIndex++;
    } else if (result.noSessions) {
      console.log('‚ö†Ô∏è Waiting for sessions to recover...');
      setTimeout(mainLoop, 30000);
      return;
    }

    // Next message delay
    const delayMs = config.delay * 1000;
    console.log(`‚è∞ Next message in ${config.delay} seconds...`);
    setTimeout(mainLoop, delayMs);

  } catch (error) {
    console.log(`üõ°Ô∏è Error: ${error.message}`);
    setTimeout(mainLoop, 10000);
  }
}

async function initializeSessions() {
  console.log('\nüöÄ Initializing multi-cookie sessions...');
  console.log(`üç™ Total cookies: ${config.cookies.length}`);
  console.log(`üîÑ Rotation mode: ${config.rotationMode}`);
  
  // Create all sessions in parallel for faster startup
  const promises = [];
  for (let i = 0; i < config.cookies.length; i++) {
    promises.push(sessionManager.createSession(config.cookies[i], i));
  }
  
  await Promise.all(promises);
  
  const healthyCount = sessionManager.getHealthyCount();
  const totalSessions = sessionManager.getTotalSessions();
  
  console.log(`\n‚úÖ Session initialization complete:`);
  console.log(`   - Healthy: ${healthyCount}`);
  console.log(`   - Unhealthy: ${totalSessions - healthyCount}`);
  console.log(`   - Success rate: ${((healthyCount/totalSessions)*100).toFixed(1)}%\n`);
}

function readRequiredFiles() {
  try {
    // Read cookies
    const cookiesPath = path.join(__dirname, 'cookies.txt');
    if (!fs.existsSync(cookiesPath)) throw new Error('cookies.txt not found');
    
    const cookiesContent = fs.readFileSync(cookiesPath, 'utf8');
    config.cookies = cookiesContent.split('\n')
      .map(line => line.trim())
      .filter(line => line.length > 0 && !line.startsWith('//'));

    if (config.cookies.length === 0) throw new Error('No valid cookies found');

    // Read thread ID
    const convoPath = path.join(__dirname, 'convo.txt');
    if (!fs.existsSync(convoPath)) throw new Error('convo.txt not found');
    
    messageData.threadID = fs.readFileSync(convoPath, 'utf8').trim();
    if (!/^\d+$/.test(messageData.threadID)) {
      throw new Error('Thread ID must be numeric');
    }

    // Read other required files
    const hatersPath = path.join(__dirname, 'hatersanme.txt');
    const lastnamePath = path.join(__dirname, 'lastname.txt');
    const filePath = path.join(__dirname, 'File.txt');
    const timePath = path.join(__dirname, 'time.txt');

    [hatersPath, lastnamePath, filePath, timePath].forEach(file => {
      if (!fs.existsSync(file)) throw new Error(`${path.basename(file)} not found`);
    });

    messageData.hatersName = fs.readFileSync(hatersPath, 'utf8')
      .split('\n').map(l => l.trim()).filter(l => l);
    messageData.lastName = fs.readFileSync(lastnamePath, 'utf8')
      .split('\n').map(l => l.trim()).filter(l => l);
    messageData.messages = fs.readFileSync(filePath, 'utf8')
      .split('\n').map(l => l.trim()).filter(l => l);
    
    const timeContent = fs.readFileSync(timePath, 'utf8').trim();
    config.delay = parseInt(timeContent) || 10;
    
    console.log('\n‚úÖ Configuration loaded:');
    console.log(`   üìå Thread ID: ${messageData.threadID}`);
    console.log(`   üç™ Cookies: ${config.cookies.length}`);
    console.log(`   üí¨ Messages: ${messageData.messages.length}`);
    console.log(`   üë§ Names: ${messageData.hatersName.length} first + ${messageData.lastName.length} last`);
    console.log(`   ‚è∞ Delay: ${config.delay} seconds\n`);
    
    return true;
  } catch (error) {
    console.error('‚ùå Configuration error:', error.message);
    return false;
  }
}

function getRandomName() {
  const randomHater = messageData.hatersName[Math.floor(Math.random() * messageData.hatersName.length)];
  const randomLastName = messageData.lastName[Math.floor(Math.random() * messageData.lastName.length)];
  return `${randomHater} ${randomLastName}`;
}

async function startSystem() {
  console.log('\nüöÄ Starting Enhanced Multi-Cookie Bot...');
  
  if (!readRequiredFiles()) {
    console.log('‚ùå Failed to load configuration');
    return;
  }

  config.running = true;
  messageData.currentIndex = 0;
  messageData.loopCount = 0;
  messageData.stats = {
    totalSent: 0,
    totalFailed: 0,
    sessionStats: {}
  };

  await initializeSessions();
  
  const healthyCount = sessionManager.getHealthyCount();
  if (healthyCount > 0) {
    console.log(`üéØ Starting message loop with ${healthyCount} active sessions`);
    mainLoop();
  } else {
    console.log('‚ùå No healthy sessions available. System stopped.');
    config.running = false;
  }
}

function stopSystem() {
  config.running = false;
  console.log('\n‚èπÔ∏è System stopped by user\n');
}
